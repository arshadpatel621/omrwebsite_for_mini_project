OMR Evaluation System

Solved Answer Sheet

Student Answer Sheet

Developed by OMR System | Node.js + HTML + CSS + JS
`); }); // ---------- PROCESS & RESULT PAGE ---------- app.post("/evaluate", (req, res) => { const { correctAnswers, studentAnswers } = req.body; const correct = correctAnswers.trim().split(","); const student = studentAnswers.trim().split(","); let total = correct.length; let score = 0; let results = []; for (let i = 0; i < total; i++) { const isCorrect = correct[i]?.trim().toUpperCase() === student[i]?.trim().toUpperCase(); if (isCorrect) score++; results.push({ question: i + 1, correct: correct[i] || "-", student: student[i] || "-", status: isCorrect ? "Correct" : "Wrong", }); } res.send(` OMR Result

OMR Evaluation Result

${results.map(r => ` `).join("")}
QuestionCorrect AnswerStudent AnswerStatus
${r.question} ${r.correct} ${r.student} ${r.status}

Go Back `); }); // ---------- START SERVER ---------- const PORT = 3000; app.listen(PORT, () => console.log(\`âœ… Server running on http://localhost:\${PORT}\`));
